# Fractional Knapsack using Greedy Method

n = int(input("Enter number of items: "))
items = []

for i in range(n):
    value = int(input(f"Enter value of item {i+1}: "))
    weight = int(input(f"Enter weight of item {i+1}: "))
    ratio = value / weight
    items.append((ratio, value, weight))

capacity = int(input("Enter knapsack capacity: "))

# Sort items by value/weight ratio in descending order
items.sort(reverse=True)

total_value = 0
used_capacity = 0

print("\nItem\tValue\tWeight\tTaken")
for ratio, value, weight in items:
    if capacity == 0:
        break
    if weight <= capacity:
        total_value += value
        capacity -= weight
        used_capacity += weight
        print(f"{value//weight}\t{value}\t{weight}\t1.0")
    else:
        fraction = capacity / weight
        total_value += value * fraction
        used_capacity += capacity
        print(f"{value//weight}\t{value}\t{weight}\t{fraction:.2f}")
        capacity = 0

print(f"\nKnapsack Capacity Used: {used_capacity}")
print(f"Maximum Value in Knapsack: {total_value:.2f}")
